一键安装qb
curl -sSL https://raw.githubusercontent.com/danielzi/script-lib/main/qbittorrent_install.sh | bash

curl -sSL https://raw.githubusercontent.com/danielzi/script-lib/main/multi_rclone_mount.sh | bash
